Table pool
Pool of samples, equivalent to a single lane for sequencing
* pool_id bigserial
  run_id bigint
* name varchar( 100 )
* created_on timestamp DEFO current_timestamp
* created_by bigint
* finalized bool DEFO 'F'
  finalized_on timestamp
* finalized_by bigint
Indexes
 idx_pool ON run_id
 idx_pool_0 ON created_by
 idx_pool_1 ON finalized_by
Uidx_pool_2 ON name
Pkpk_pool ON pool_id
Foreign Keys
 fk_pool ( run_id ) ref run (run_id)
 fk_pool_0 ( created_by ) ref people (person_id)
 fk_pool_1 ( finalized_by ) ref people (person_id)
